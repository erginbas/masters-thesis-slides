% Staged version of the FASMT bin-splitting tree for progressive reveal.
% Produces one page per stage (7 pages total), following the depth-first
% (lexicographic) processing order.
% Running example (n = 4): F(0100)=2, F(0001)=3, F(0101)=1.
% Adaptive test vectors: h_eps = 1011, h_0 = 0100, h_1 = 1010, h_10 = 0100.

\usepackage{tikz}
\usetikzlibrary{arrows.meta,positioning,calc,fit,backgrounds}
\usepackage{amsmath,amssymb}
\usepackage{forest}
\usepackage{xcolor}

\newcommand{\bh}{\mathbf{h}}
\newcommand{\bk}{\mathbf{k}}
\newcommand{\bell}{\boldsymbol{\ell}}
\newcommand{\beps}{\boldsymbol{\epsilon}}

\definecolor{hepscolor}{RGB}{0,100,200}
\definecolor{h0color}{RGB}{200,50,50}

\definecolor{ordercolor}{RGB}{180,100,30}

\newlength{\CircleSize}
\setlength{\CircleSize}{52pt}

% Stage machinery: elements tagged app=<k> become invisible while \stage < k.
\tikzset{
  app/.code={\ifnum\stage<#1\relax
    \pgfkeysalso{/tikz/opacity=0,/tikz/text opacity=0}\fi},
  binlabel/.style={font=\large\bfseries, fill=yellow!30, rounded corners=2pt, inner sep=3pt, xshift=-8pt, yshift=-4pt},
  binlabelpruned/.style={font=\large\bfseries, fill=black!8, text=black!50, rounded corners=2pt, inner sep=3pt, xshift=-8pt, yshift=-4pt},
  peelbox/.style={font=\large, draw=orange!70!black, rounded corners=4pt, fill=orange!20, inner sep=6pt, minimum width=70pt, minimum height=50pt, line width=1pt},
  peelconnector/.style={very thick, densely dotted, orange!70!black, -Latex},
  orderlabel/.style={font=\normalsize\bfseries, text=ordercolor, circle, draw=ordercolor, fill=ordercolor!10, inner sep=2pt, minimum size=20pt, line width=0.8pt, xshift=3pt, yshift=-4pt},
}
